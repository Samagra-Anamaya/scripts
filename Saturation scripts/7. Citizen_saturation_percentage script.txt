------------------------------------------------------------------------------------------------------------------------------------
drop table if exists citizen_saturation_percentage;
create table citizen_saturation_percentage
as
select all_cols.column_name,coalesce((b.total_cnt/a.total_cnt)*100,0) as saturation_percent, coalesce(b.total_cnt,0) as total_saturation_count, coalesce(a.total_cnt,0) as total_null_count from 

(select distinct column_name from citizen_saturation_counts) all_cols
left join
(SELECT sum(count) as total_cnt,column_name FROM citizen_saturation_counts where (source_id <> 0 or source_id is null)
group by column_name)a on all_cols.column_name = a.column_name
left join
(SELECT sum(count) as total_cnt,column_name FROM citizen_saturation_counts where source_id <> 0 and source_id is not null
group by column_name)b on a.column_name = b.column_name
order by all_cols.column_name;
